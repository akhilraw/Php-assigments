1. create database
	statement:
	
	CREATE DATABASE programming_test_php;

2. create table agent_master
	statement:
	
	CREATE TABLE Agent_Master
	(Agent_No INT AUTO_INCREMENT PRIMARY KEY,
	Agent_Name VARCHAR(25) NOT NULL);

3. create table policy_details
	statement:

	CREATE TABLE Policy_Details
	(Policy_No INT AUTO_INCREMENT PRIMARY KEY,
	Agent_No INT,
	PolicyDate DATE NOT NULL,
	CustomerName VARCHAR(25) NOT NULL,
	PolicyAmount INT NOT NULL,
	Commission DECIMAL(15,2));

4. create final display_report
	statement:

	SELECT a.Agent_Name AS Agent, p.PolicyDate, p.PolicyAmount, p.Commission
	FROM policy_details AS p
	LEFT JOIN agent_master AS a ON p.Agent_No=a.Agent_No;

5. join first and second table
	statement:

	SELECT * FROM agent_master
	NATURAL JOIN policy_details;
